def secrets=[[
            path:'jenkins/jenkintest',
            secretValues:[envVar:'vault_sec',vaultKey:'test']
            ]]
def config=[
            $class :'VaultAppRoleCredential'
            vaultUrl:'http://192.168.33.20:8200',
            vaultCredentialId:'vault'
            ]
    
pipeline{
    agent any
    stages{
        stage('build stage'){
            steps{
                echo "We are building the application"
                bat "Go build -o hello.exe hello.go"
                bat "powershell.exe Compress-Archive  hello.exe Hello-1.0.%BUILD_NUMBER%.zip"
            }
        }
    }
    post{
        always{
            echo "The job is completed"
        }
        success{
            withVault([configuration: config,vaultSecrets: secrets]){
                bat "echo %vault_sec%"
            }
            echo "success"
            bat "jfrog rt c rt-server-1 --url=http://192.168.33.20:8082/artifactory --user=admin --password=Prasanth "
            bat "jfrog rt u Hello-1.0.%BUILD_NUMBER%.zip CodeBuildtest"
        }
        unstable{
            echo "Build unstable"
        }
        failure{
            echo "Build failed"
        }
        changed{
            echo "Build changed"
        }
    }
}

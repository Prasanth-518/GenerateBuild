pipeline{
    agent any
    stages{
        stage('build stage'){
            steps{
                echo "We are building the application"
                bat "Go build hello.go"
            }
        }
    }
    post{
        always{
            echo "The job is completed"
        }
        success{
            echo "success"
        }
        unstable{
            echo "Build unstable"
        }
        failure{
            echo "Build failed"
        }
        changed{
            echo "Build changed"
        }
    }
}

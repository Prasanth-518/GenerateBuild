pipeline{
    agent any
    stages{
        stage('build stage'){
            steps{
                echo "We are building the application"
                bat "Go build -o hello1.exe hello.go"
                bat "powershell.exe Compress-Archive hello1.exe Hello1.zip"
            }
        }
    }
    post{
        always{
            echo "The job is completed"
        }
        success{
            echo "success"
            bat "jfrog rt c rt-server-1 --url=http://192.168.33.20:8082 --user=admin --password=Prasanth "
            bat "jfrog rt u Hello.zip CodeBuildTest"
        }
        unstable{
            echo "Build unstable"
        }
        failure{
            echo "Build failed"
        }
        changed{
            echo "Build changed"
        }
    }
}
